##### (C) RSI 14일 점수 (0-5점)

```typescript
// Wilder's Smoothing RSI 계산
function calculateRSI(prices: number[], period = 14): number {
  // 초기 평균 이득/손실 (SMA)
  let avgGain = 0, avgLoss = 0;
  for (let i = 1; i <= period; i++) {
    const change = prices[i] - prices[i - 1];
    if (change > 0) avgGain += change;
    else avgLoss += Math.abs(change);
  }
  avgGain /= period;
  avgLoss /= period;

  // Wilder Smoothing (나머지 데이터)
  for (let i = period + 1; i < prices.length; i++) {
    const change = prices[i] - prices[i - 1];
    avgGain = (avgGain * (period - 1) + (change > 0 ? change : 0)) / period;
    avgLoss = (avgLoss * (period - 1) + (change < 0 ? Math.abs(change) : 0)) / period;
  }

  if (avgLoss === 0) return 100;
  const rs = avgGain / avgLoss;
  return 100 - (100 / (1 + rs));
}
```

| RSI 범위 | 점수 | 상태 |
|----------|------|------|
| < 30 | 5 | 과매도 (매수 기회) |
| 30-40 | 4 | 약세 |
| 40-50 | 3 | 중립 약세 |
| 50-60 | 2 | 중립 강세 |
| 60-70 | 1 | 강세 |
| > 70 | 0 | 과매수 (매도 고려) |

##### (D) MACD 신호 (Phase 2 확장)

```typescript
interface MACDResult {
  macdLine: number;      // EMA(12) - EMA(26)
  signalLine: number;    // EMA(9) of MACD
  histogram: number;     // MACD - Signal
  crossover: 'bullish' | 'bearish' | 'none';
}

function calculateMACD(prices: number[]): MACDResult {
  const ema12 = calculateEMA(prices, 12);
  const ema26 = calculateEMA(prices, 26);
  const macdLine = ema12 - ema26;
  
  // MACD 히스토리 계산 후 Signal Line
  const macdHistory = prices.map((_, i) => {
    if (i < 25) return 0;
    return calculateEMA(prices.slice(0, i + 1), 12) - 
           calculateEMA(prices.slice(0, i + 1), 26);
  });
  const signalLine = calculateEMA(macdHistory.slice(-9), 9);
  
  return {
    macdLine,
    signalLine,
    histogram: macdLine - signalLine,
    crossover: detectCrossover(macdHistory, signalLine)
  };
}
```

| MACD 조건 | 점수 가산 | 설명 |
|-----------|-----------|------|
| 골든크로스 (MACD > Signal 전환) | +2 | 상승 전환 신호 |
| 데드크로스 (MACD < Signal 전환) | -2 | 하락 전환 신호 |
| 히스토그램 양전환 | +1 | 모멘텀 개선 |
| 히스토그램 음전환 | -1 | 모멘텀 악화 |

##### (E) M/W 패턴 감지 (0-5점)

```typescript
/**
 * 패턴 감지 알고리즘:
 * 1. 최근 60일 가격에서 극값(local min/max) 추출
 * 2. 극값 패턴으로 W(이중바닥) / M(이중천장) 판별
 * 3. 넥라인 돌파 여부로 확인
 */
function detectMWPattern(prices: number[]): PatternResult {
  const pivots = findPivotPoints(prices, 5); // ±5일 윈도우
  
  // W패턴: Low → High → Low(비슷) → High(넥라인 돌파)
  if (isDoubleBottom(pivots)) {
    return { score: 5, pattern: 'accumulation', confidence: 0.8 };
  }
  
  // M패턴: High → Low → High(비슷) → Low(넥라인 이탈)
  if (isDoubleTop(pivots)) {
    return { score: 0, pattern: 'distribution', confidence: 0.8 };
  }
  
  // 하락 추세 중 → 잠재적 바닥 형성 기대
  const trend = (prices[prices.length - 1] - prices[0]) / prices[0];
  if (trend < -0.05) return { score: 4, pattern: 'accumulation', confidence: 0.5 };
  if (trend > 0.05) return { score: 1, pattern: 'distribution', confidence: 0.5 };
  return { score: 3, pattern: 'neutral', confidence: 0.3 };
}
```

### 2.2 주가 판독기 (Valuation Screener)

#### 2.2.1 설계 철학

> "적정주가를 알아야 비싼지 싼지 판단할 수 있다" — 소몽

주가 판독기는 **펀더멘털 밸류에이션** 기반으로 현재 종목의 가치 대비 가격을 0~20점으로 수치화한다.

#### 2.2.2 핵심 공식 체계

```
┌─────────────────────────────────────────────────────────┐
│                 소몽의 적정주가 공식                      │
│                                                         │
│  [1] 적정 PER = EPS 성장률(%) 중앙값                    │
│      → 5년 Consensus EPS 성장률의 median                │
│                                                         │
│  [2] 예상 EPS = 현재 EPS × (1 + 내년 성장률%)           │
│                                                         │
│  [3] 적정주가 = 예상 EPS × 적정 PER                     │
│                                                         │
│  [4] 상승여력(%) = (적정주가 - 현재가) / 현재가 × 100   │
│                                                         │
│  [5] PEG = PER / EPS 성장률                             │
│      → PEG < 1: 성장 대비 저평가                        │
│      → PEG = 1: 적정                                    │
│      → PEG > 1: 성장 대비 고평가                        │
└─────────────────────────────────────────────────────────┘
```

##### (A) 시장 PER 대비 종목 PER 점수 (0-5점)

```typescript
function calculatePEScore(stockPE: number, marketPE: number): number {
  if (stockPE <= 0) return 0; // 적자 기업 → 이 지표 무효
  const ratio = stockPE / marketPE;
  
  // 종목 PER이 시장 PER 대비 낮을수록 유리
  if (ratio < 0.5) return 5;   // 시장 PER의 절반 이하
  if (ratio < 0.75) return 4;
  if (ratio < 1.0) return 3;   // 시장 PER과 비슷
  if (ratio < 1.25) return 2;
  if (ratio < 1.5) return 1;
  return 0;                     // 시장 PER의 1.5배 이상
}
```

##### (B) 적정주가 대비 현재가 점수 (0-5점)

```typescript
function calculateFairPriceScore(currentPrice: number, fairPrice: number): number {
  if (fairPrice <= 0) return 0;
  const upside = ((fairPrice - currentPrice) / currentPrice) * 100;
  
  if (upside >= 50) return 5;  // 50% 이상 상승 여력
  if (upside >= 30) return 4;
  if (upside >= 10) return 3;
  if (upside >= 0) return 2;   // 적정가 근처
  if (upside >= -15) return 1;
  return 0;                     // 15% 이상 고평가
}
```

### 2.3 역발상 판독기 (Sentiment/Contrarian Screener)

#### 2.3.1 설계 철학

> "남들이 공포에 떨 때 매수하고, 남들이 환호할 때 매도하라" — 소몽 (워런 버핏)

역발상 판독기는 **시장 전체의 심리 상태**를 5개 지표로 종합하여 0~25점으로 수치화한다. 종목 개별 지표가 아닌 **시장 레벨** 지표.

  chartScore: number;      // 0-20
  valuationScore: number;  // 0-20
  sentimentScore: number;  // 0-25
  rawTotal: number;        // 0-65
  normalizedScore: number; // 0-100 (rawTotal / 65 × 100)

  confidence: number;      // 0-1
  positionSize: number;    // 권장 비중 (%)
}

function generateCombinedSignal(
  chart: ChartResult,
  valuation: ValuationResult,
  sentiment: SentimentResult
): CombinedSignal {
  const rawTotal = chart.scores.total + valuation.scores.total + sentiment.totalScore;
  const normalized = Math.round((rawTotal / 65) * 100);
  
  let signal: CombinedSignal['signal'];
  let positionSize: number;
  
  if (normalized >= 80) {
    signal = 'strong_buy';
    positionSize = 30; // 포트폴리오의 30%까지
  } else if (normalized >= 60) {
    signal = 'buy';
    positionSize = 20;
  } else if (normalized >= 40) {
    signal = 'hold';
    positionSize = 0;  // 추가 매수 없음
  } else if (normalized >= 20) {
    signal = 'sell';
    positionSize = -20; // 20% 비중 축소
  } else {
    signal = 'strong_sell';
    positionSize = -50; // 50% 비중 축소
  }
  
  // 신뢰도: 3개 판독기가 일치할수록 높음
  const verdicts = [chart.verdict, valuation.verdict, sentiment.verdict];
  const agreement = calculateAgreement(verdicts);
  
  return {
    chartScore: chart.scores.total,
    valuationScore: valuation.scores.total,
    sentimentScore: sentiment.totalScore,
    rawTotal,
    normalizedScore: normalized,
    signal,
    confidence: agreement,

import type { ChartData, FinancialData } from '@/types';
import { calculateRSI, calculateMA } from '../screeners/chart-screener';

async function getYF() {
  const yf = await import('yahoo-finance2');
  return yf.default;
}

/**
 * 차트 데이터 수집
 * - 300일 일봉 → 200일선 계산에 필요
 * - RSI 14일 계산
 * - 최근 60일 가격 (패턴 분석용)
 */
export async function fetchChartData(ticker: string): Promise<ChartData> {
  const yf = await getYF();
  
  const endDate = new Date();
  const startDate = new Date();
  startDate.setDate(startDate.getDate() - 300);

  const result = await yf.historical(ticker, {
    period1: startDate,
    period2: endDate,
    interval: '1d',
  });

  const closes = result.map(r => r.close);
  const latest = result[result.length - 1];

  return {
    ticker,
    date: latest.date.toISOString().split('T')[0],
    close: latest.close,
    ma50: Math.round(calculateMA(closes, 50) * 100) / 100,
    ma200: Math.round(calculateMA(closes, 200) * 100) / 100,
    rsi14: Math.round(calculateRSI(closes) * 100) / 100,
    volume: latest.volume,
    prices: closes.slice(-60),
  };
}

/**
 * 재무 데이터 수집
 * quoteSummary의 earningsTrend에서 EPS 성장률 추출
 */
export async function fetchFinancialData(ticker: string): Promise<FinancialData> {
  const yf = await getYF();

  const [quote, summary] = await Promise.all([
    yf.quote(ticker),
    yf.quoteSummary(ticker, {
      modules: ['defaultKeyStatistics', 'earningsTrend', 'financialData'],
    }),
  ]);

  const trends = summary.earningsTrend?.trend ?? [];
  const epsGrowthRates = trends
    .filter(t => t.growth != null)
    .map(t => t.growth! * 100);

  // 폴백: PEG에서 역산
  if (epsGrowthRates.length === 0) {
    const pegRatio = summary.defaultKeyStatistics?.pegRatio;
    if (pegRatio && pegRatio > 0) {
      epsGrowthRates.push((quote.trailingPE ?? 20) / pegRatio);
    } else {
      epsGrowthRates.push(10); // 보수적 기본값
    }
  }

  // S&P500 PER
  let marketPE = 22;
  try {
    const spy = await yf.quote('SPY');
    marketPE = spy.trailingPE ?? 22;
  } catch { /* 기본값 사용 */ }

  return {
    ticker,
    currentPrice: quote.regularMarketPrice ?? 0,
    currentEPS: quote.epsTrailingTwelveMonths ?? 0,
    peRatio: quote.trailingPE ?? 0,
    forwardPE: quote.forwardPE ?? 0,
    epsGrowthRates,
    marketPE,
  };
}

// Redis 기반 Rate Limiter
async function rateLimitedFetch<T>(
  key: string,
  fetchFn: () => Promise<T>,
  cacheTTL: number = 900 // 15분
): Promise<T> {
  // 1. 캐시 확인
  const cached = await redis.get(key);
  if (cached) return JSON.parse(cached);
  
  // 2. Rate limit 확인
  const count = await redis.incr(`ratelimit:${key.split(':')[0]}`);
  if (count === 1) await redis.expire(`ratelimit:${key.split(':')[0]}`, 60);
  if (count > RATE_LIMITS.yahoo.maxPerMinute) {
    throw new Error('Rate limit exceeded');
  }
  
  // 3. 실제 API 호출
  const result = await fetchFn();
  
  // 4. 캐시 저장
  await redis.set(key, JSON.stringify(result), { ex: cacheTTL });
  
  return result;
}

export async function GET(request: Request) {
  // Vercel Cron 인증
  const authHeader = request.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return Response.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const TRACKED_TICKERS = await getTrackedTickers(); // 사용자들이 추적 중인 종목
  const BATCH_SIZE = 10;
  const DELAY_MS = 2000; // 배치 간 2초 대기
  
  const results = [];
  
  for (let i = 0; i < TRACKED_TICKERS.length; i += BATCH_SIZE) {
    const batch = TRACKED_TICKERS.slice(i, i + BATCH_SIZE);
    
    const batchResults = await Promise.allSettled(
      batch.map(async (ticker) => {
        const [chart, financial] = await Promise.all([
          fetchChartData(ticker),
          fetchFinancialData(ticker),
        ]);
        
        const chartScore = calculateChartScore(chart);
        const valuationScore = calculateValuationScore(financial);
        
        // Supabase 저장
        await supabase.from('screener_scores').upsert({
          ticker,
          date: new Date().toISOString().split('T')[0],
          ...flattenScores(chartScore, valuationScore),
        });
        
        return { ticker, status: 'success' };
      })
    );
    
    results.push(...batchResults);
    
    // Rate limit 준수
    if (i + BATCH_SIZE < TRACKED_TICKERS.length) {
      await new Promise(resolve => setTimeout(resolve, DELAY_MS));
    }
  }
  
  // 센티먼트 (시장 전체 → 1회만)
  const sentiment = await fetchSentimentData();
  const sentimentScore = calculateSentimentScore(sentiment);
  await supabase.from('market_sentiment').upsert({
    date: new Date().toISOString().split('T')[0],
    ...sentimentScore,
  });
  
  return Response.json({ updated: results.length, timestamp: new Date() });

#### 4.1.2 신호 생성 알고리즘

```typescript
interface SignalOutput {
  signal: 'strong_buy' | 'buy' | 'hold' | 'sell' | 'strong_sell';
  confidence: number;
  positionSizePct: number;
  reasoning: string[];
  riskLevel: 'low' | 'medium' | 'high' | 'extreme';
  timeHorizon: 'short' | 'medium' | 'long';
}

function generateSignal(
  chart: ChartResult,
  valuation: ValuationResult,
  sentiment: SentimentResult,
  liquidity: LiquidityCheck,
  history: ScreenerHistory[]
): SignalOutput {
  
  // ─── Step 1: 기본 점수 합산 ───
  const rawScore = chart.scores.total + valuation.scores.total + sentiment.totalScore;
  const maxScore = 20 + 20 + 25; // 65
  const adjustedScore = rawScore * (1 + liquidity.adjustment);
  const normalizedScore = Math.round((adjustedScore / maxScore) * 100);
  
  // ─── Step 2: 컨센서스 체크 ───
  // 3개 판독기가 같은 방향을 가리킬수록 신뢰도 UP
  const directions = [
    chart.scores.total > 10 ? 'bullish' : chart.scores.total < 7 ? 'bearish' : 'neutral',
    valuation.scores.total > 10 ? 'bullish' : valuation.scores.total < 7 ? 'bearish' : 'neutral',
    sentiment.totalScore > 15 ? 'bullish' : sentiment.totalScore < 8 ? 'bearish' : 'neutral',
  ];
  
  const bullishCount = directions.filter(d => d === 'bullish').length;
  const bearishCount = directions.filter(d => d === 'bearish').length;
  const consensus = Math.max(bullishCount, bearishCount) / 3;
  
  // ─── Step 3: 신호 변화 감지 ───
  const prevScore = history.length > 0 ? history[0].normalizedScore : normalizedScore;
  const scoreChange = normalizedScore - prevScore;
  const isMomentumShift = Math.abs(scoreChange) > 15; // 15% 이상 변화
  
  // ─── Step 4: 최종 신호 결정 ───
  let signal: SignalOutput['signal'];
  let positionSizePct: number;
  const reasoning: string[] = [];
  
  if (normalizedScore >= 75) {
    signal = 'strong_buy';
    positionSizePct = 25 + (consensus * 10); // 25-35%
    reasoning.push(`종합 점수 ${normalizedScore}/100 — 3종 판독기 모두 매수 신호`);
  } else if (normalizedScore >= 55) {
    signal = 'buy';
    positionSizePct = 10 + (consensus * 10); // 10-20%
    reasoning.push(`종합 점수 ${normalizedScore}/100 — 매수 유리 구간`);
  } else if (normalizedScore >= 40) {
    signal = 'hold';
    positionSizePct = 0;
    reasoning.push(`종합 점수 ${normalizedScore}/100 — 관망 구간`);
  } else if (normalizedScore >= 25) {
    signal = 'sell';
    positionSizePct = -(10 + (consensus * 10));
    reasoning.push(`종합 점수 ${normalizedScore}/100 — 비중 축소 권고`);
  } else {
    signal = 'strong_sell';
    positionSizePct = -(25 + (consensus * 10));
    reasoning.push(`종합 점수 ${normalizedScore}/100 — 적극 매도 권고`);
  }
  
  // 개별 판독기 근거 추가
  if (chart.verdict === 'fear') {
    reasoning.push(`차트: RSI ${chart.signals.rsiLevel}, 이격도 ${chart.signals.deviationPct}% — 기술적 과매도`);
  }
  if (valuation.verdict === 'undervalued') {
    reasoning.push(`밸류: 적정주가 $${valuation.fairPrice} (상승여력 ${valuation.upsideDownside}%)`);
  }
  if (sentiment.verdict === 'extreme_fear' || sentiment.verdict === 'fear') {
    reasoning.push(`심리: VIX ${sentiment.vix.current}, 시장 공포 상태 — 역발상 매수 기회`);
  }
  
  // 모멘텀 시프트 경고
  if (isMomentumShift) {
    reasoning.push(`⚠️ 최근 신호 급변 (${scoreChange > 0 ? '+' : ''}${scoreChange}점) — 추세 전환 가능성`);
  }
  
  return {
    signal,
    confidence: consensus,
    positionSizePct,
    reasoning,
    riskLevel: calculateRiskLevel(chart, valuation, sentiment),
    timeHorizon: determineTimeHorizon(chart, valuation),
  };
}

    score: volatilityRisk,
    weight: 0.25,
    description: `RSI ${chart.signals.rsiLevel}, 이격도 ${chart.signals.deviationPct}%`,
  });
  
  // ─── 2. 밸류에이션 리스크 (25%) ───
  const valuationRisk = calculateValuationRisk(valuation);
  factors.push({
    name: '밸류에이션',
    score: valuationRisk,
    weight: 0.25,
    description: `PER ${valuation.peRatio}x, PEG ${valuation.peg}`,
  });
  
  // ─── 3. 시장 리스크 (30%) ───
  const marketRisk = calculateMarketRisk(sentiment);
  factors.push({
    name: '시장 환경',
    score: marketRisk,
    weight: 0.30,
    description: `VIX ${sentiment.vix.current}, HY Spread ${sentiment.hySpread.current}%`,
  });
  
  // ─── 4. 집중도 리스크 (20%) ───
  const concentrationRisk = portfolioContext 
    ? calculateConcentrationRisk(portfolioContext)
    : 12; // 기본값
  factors.push({
    name: '집중도',
    score: concentrationRisk,
    weight: 0.20,
    description: portfolioContext 
      ? `포트폴리오 내 비중 ${portfolioContext.weightPct}%`
      : '포트폴리오 미등록',
  });
  
  // 가중 합산
  const riskScore = Math.round(
    factors.reduce((sum, f) => sum + f.score * f.weight, 0)
  );
  
  let overallRisk: RiskAssessment['overallRisk'];
  if (riskScore >= 75) overallRisk = 'extreme';
  else if (riskScore >= 50) overallRisk = 'high';
  else if (riskScore >= 25) overallRisk = 'medium';
  else overallRisk = 'low';
  
  // 예상 최대 손실률 (VIX 기반 간이 추정)
  const maxDrawdown = estimateMaxDrawdown(sentiment.vix.current, volatilityRisk);
  
  // 켈리 기준 포지션 사이즈 (간소화)
  const positionSize = calculateKellyPosition(riskScore, valuation.upsideDownside);
  
  return {
    overallRisk,
    riskScore,
    factors,
    maxDrawdownEstimate: maxDrawdown,
    positionSizeRecommendation: positionSize,
  };
}

// 변동성 리스크
function calculateVolatilityRisk(chart: ChartResult): number {
  let risk = 0;
  // RSI 극단값 → 높은 변동성
  if (chart.signals.rsiLevel === 'overbought' || chart.signals.rsiLevel === 'oversold') risk += 15;
  // 이격도 큼 → 평균 회귀 리스크
  if (Math.abs(chart.signals.deviationPct) > 10) risk += 10;
  // 이평선 하회 → 하락 추세 리스크
  if (chart.signals.maStatus === 'below_both') risk += 15;
  return Math.min(risk, 25);
}

// 밸류에이션 리스크
function calculateValuationRisk(valuation: ValuationResult): number {
  let risk = 0;
  if (valuation.peRatio > 40) risk += 10;        // 고PER
  if (valuation.peg > 2) risk += 8;              // 성장 대비 고평가
  if (valuation.upsideDownside < -20) risk += 12; // 적정가 대비 20% 이상 고평가
  return Math.min(risk, 25);
}

// 시장 리스크
function calculateMarketRisk(sentiment: SentimentResult): number {
  // 역발상: 시장 공포 = 매수 기회이지만, 리스크는 높음
  let risk = 0;
  if (sentiment.vix.current > 30) risk += 10;
  if (sentiment.hySpread.current > 5) risk += 8;
  if (sentiment.verdict === 'extreme_greed') risk += 12; // 버블 리스크
  return Math.min(risk, 25);
}

// 예상 최대 손실률
function estimateMaxDrawdown(vix: number, volatilityRisk: number): number {
  // VIX ≈ 연간 변동성의 기대값
  // 최대 손실 ≈ VIX × 2 (2표준편차, 95% 신뢰구간)
  const annualizedVol = vix / 100;
  const monthlyVol = annualizedVol / Math.sqrt(12);
  return Math.round(monthlyVol * 200); // 2σ, 1개월 기준
}

// 켈리 기준 포지션 사이즈 (간소화)
function calculateKellyPosition(riskScore: number, expectedReturn: number): number {
  // 풀 켈리는 너무 공격적 → 하프 켈리 사용
  const winProb = Math.max(0.3, 1 - riskScore / 100);
  const kelly = winProb - ((1 - winProb) / (expectedReturn / 100 || 1));
  const halfKelly = kelly / 2;
  return Math.max(0, Math.min(30, Math.round(halfKelly * 100))); // 0-30% 제한
}
```

## 10. 부록

### 10.1 핵심 공식 요약

```
[적정주가] = 예상 EPS × 적정 PER
[적정 PER] = EPS 성장률(%) 중앙값
[예상 EPS] = 현재 EPS × (1 + 내년 성장률%)
[PEG] = PER / EPS 성장률
[이격도] = (현재가 - 50일선) / 50일선 × 100
[RSI] = 100 - 100/(1 + RS)  where RS = AvgGain/AvgLoss
[MACD] = EMA(12) - EMA(26), Signal = EMA(9) of MACD
```

### 10.2 점수 체계 종합

```
차트 판독기:   0-20점 (4지표 × 5점)
주가 판독기:   0-20점 (4지표 × 5점)
역발상 판독기: 0-25점 (5지표 × 5점)
───────────────────────────
종합 원점수:   0-65점
정규화 점수:   0-100점 (= rawTotal / 65 × 100)

※ 모든 점수: 높을수록 = 공포 = 역발상 매수 기회
```
